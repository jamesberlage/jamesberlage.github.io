(defn factorial
  [x]
  (cond
    (not= (type x) js/Number) (throw (js/Error. "x must be a number"))
    (< x 0)                   (throw (js/Error. "x must be > 0"))
    (= x 0)                   1
    :else                     (* x (factorial (- x 1)))))
